import re, os

pages = {
    'docs/index.html': 'index.html',
    'docs/features.html': 'features.html',
    'docs/walkthroughs.html': 'walkthroughs.html',
    'docs/getting-started.html': 'getting-started.html',
}

errors = []
for filepath, filename in pages.items():
    html = open(filepath).read()
    # Check no absolute asset paths
    if re.search(r'(src|href)=\"/assets/', html):
        errors.append(f'{filename}: absolute asset path found')
    # Check nav links exist
    for target in ['index.html', 'features.html', 'walkthroughs.html', 'getting-started.html']:
        if f'href=\"{target}\"' not in html:
            errors.append(f'{filename}: missing nav link to {target}')
    # Check no localhost links
    if 'localhost' in html or '127.0.0.1' in html:
        errors.append(f'{filename}: localhost reference found')
    # Check all referenced images exist
    for img_match in re.findall(r'src=\"(assets/images/[^\"]+)\"', html):
        img_path = os.path.join('docs', img_match)
        if not os.path.exists(img_path):
            errors.append(f'{filename}: missing image {img_match}')

if errors:
    for e in errors:
        print(f'FAIL: {e}')
    raise AssertionError(f'{len(errors)} link audit errors')
else:
    print(f'All {len(pages)} pages passed link audit (nav links, asset paths, image refs)')
"
